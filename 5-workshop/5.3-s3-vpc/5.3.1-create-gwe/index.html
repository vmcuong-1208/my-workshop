Configure Cognito and API Gateway

Create a Cognito user pool

  1. Open Amazon Cognito and choose User pools.
  2. Choose Create user pool.
  3. In Define your application, choose Single-page application (SPA).
  4. Under Options for sign-in identifiers, choose Email.
  5. Under Required attributes for sign-up, choose email and name.
  6. Create the user directory.
  7. Update the password policy to require at least 10 characters, uppercase letters, and numbers.

Configure Hosted UI and Google sign-in

  1. Create a Cognito domain for Hosted UI.
  2. In Google Cloud, configure the OAuth consent screen.
  3. Create an OAuth Client ID for the Cognito Hosted UI.
  4. Add the Google client ID and secret to Cognito.
  5. Enable Google as an identity provider in Hosted UI.
  6. Configure callback and sign-out URLs:
Allowed callback URL: http://localhost:5173/auth/google/callback
Allowed sign-out URL: http://localhost:5173

Create the main Lambda API

  1. Open AWS Lambda and choose Create function.
  2. Choose Author from scratch.
  3. Function name: AI-Personal-Learning-Handle.
  4. Runtime: Node.js 22.x or the latest supported Node.js runtime available in your account.
  5. Create the function and deploy the handler code.

Use the following complete handler for index.mjs. Update BUCKET_NAME, queue URLs, and table names if you used different names.

import { DynamoDBClient } from "@aws-sdk/client-dynamodb";
import {
  DynamoDBDocumentClient,
  QueryCommand,
  GetCommand,
  PutCommand,
  DeleteCommand
} from "@aws-sdk/lib-dynamodb";
import { S3Client, PutObjectCommand, GetObjectCommand } from "@aws-sdk/client-s3";
import { getSignedUrl } from "@aws-sdk/s3-request-presigner";
import { SQSClient, SendMessageCommand } from "@aws-sdk/client-sqs";


const client = new DynamoDBClient({});
const docClient = DynamoDBDocumentClient.from(client);
const s3Client = new S3Client({ region: "ap-southeast-1" });
const sqsClient = new SQSClient({ region: "ap-southeast-1" });


// Các hằng số bảng
const TABLE_NAME = "JournalLogs";
const BUCKET_NAME = "ai-personal-learning-images-1208";
const AI_REPORT_TABLE = "AiReports";
const QUIZ_TABLE = "Quizzes";
const QUIZ_ATTEMPT_TABLE = "QuizAttempts";


// URL Hàng đợi SQS (Thay bằng URL của bạn)
const AI_REPORT_QUEUE_URL = "https://sqs.ap-southeast-1.amazonaws.com/467638031040/ai-report-jobs";
const QUIZ_QUEUE_URL = "https://sqs.ap-southeast-1.amazonaws.com/467638031040/quiz-jobs";


// Hàm hỗ trợ sinh Signed GET URL cho mảng ảnh (Giữ nguyên gốc)
async function attachSignedUrlsToImages(imagesArray) {
  if (!Array.isArray(imagesArray) || imagesArray.length === 0) return [];


  return await Promise.all(
    imagesArray.map(async (img) => {
      const imageKey = typeof img === "string" ? img : (img.image_key || img.imageKey || img.key || img.image_url);


      if (!imageKey || imageKey.startsWith("http")) {
        return typeof img === "string" ? { image_key: img, download_url: img } : { ...img, download_url: img.download_url || img.image_url };
      }


      try {
        const getCommand = new GetObjectCommand({
          Bucket: BUCKET_NAME,
          Key: imageKey,
        });
        const downloadUrl = await getSignedUrl(s3Client, getCommand, { expiresIn: 7200 });


        if (typeof img === "string") {
          return { image_key: img, download_url: downloadUrl, image_url: downloadUrl };
        }
        return {
          ...img,
          image_key: imageKey,
          download_url: downloadUrl,
          image_url: downloadUrl,
        };
      } catch (err) {
        console.error("Lỗi tạo signed GET URL:", err);
        return typeof img === "object" ? img : { image_key: img };
      }
    })
  );
}


export const handler = async (event) => {
  const headers = {
    "Content-Type": "application/json",
    "Access-Control-Allow-Origin": "*",
    "Access-Control-Allow-Headers": "Content-Type,Authorization",
    "Access-Control-Allow-Methods": "OPTIONS,POST,GET,PUT,DELETE"
  };


  const method = event.requestContext?.http?.method;
  const path = event.rawPath;
  const pathParameters = event.pathParameters || {};
  const claims = event.requestContext?.authorizer?.jwt?.claims;


  // 1. Route /health
  if (path === "/health") {
    return {
      statusCode: 200,
      headers,
      body: JSON.stringify({ ok: true, message: "Hệ thống hoạt động bình thường!" }),
    };
  }


  // Xác thực User từ Token
  const userId = claims?.sub;
  if (!userId && path !== "/health") {
    return {
      statusCode: 401,
      headers,
      body: JSON.stringify({ message: "Không tìm thấy thông tin xác thực người dùng." }),
    };
  }


  try {
    // 2. Route GET /me
    if (path === "/me" && method === "GET") {
      return {
        statusCode: 200,
        headers,
        body: JSON.stringify({
          userId,
          username: claims['cognito:username'],
          email: claims.email,
          name: claims.name || claims.given_name || "User"
        }),
      };
    }


    // 3. Route POST /logs/{id}/images/presigned-url (Tạo URL để Upload)
    if (pathParameters.id && path.endsWith("/images/presigned-url") && method === "POST") {
      const entryId = pathParameters.id;
      const body = JSON.parse(event.body || "{}");
      const fileName = body.file_name || body.fileName || "image.png";
      const contentType = body.content_type || body.contentType;
      const size = body.size;


      const allowedTypes = ["image/png", "image/jpeg", "image/webp"];
      if (!contentType || !allowedTypes.includes(contentType)) {
        return {
          statusCode: 400,
          headers,
          body: JSON.stringify({ message: "Định dạng file không hợp lệ (chỉ cho phép png, jpeg, webp)" }),
        };
      }


      if (size && size > 5242880) {
        return {
          statusCode: 400,
          headers,
          body: JSON.stringify({ message: "Dung lượng file vượt quá 5MB" }),
        };
      }


      let ext = "png";
      if (contentType === "image/jpeg") ext = "jpg";
      if (contentType === "image/webp") ext = "webp";


      const imageId = `img_${Date.now()}`;
      const imageKey = `users/${userId}/journal/${entryId}/${imageId}.${ext}`;


      const command = new PutObjectCommand({
        Bucket: BUCKET_NAME,
        Key: imageKey,
        ContentType: contentType,
      });


      const expiresInSeconds = 900;
      const uploadUrl = await getSignedUrl(s3Client, command, { expiresIn: expiresInSeconds });


      return {
        statusCode: 200,
        headers,
        body: JSON.stringify({
          upload_url: uploadUrl,
          image_key: imageKey,
          expires_in: expiresInSeconds
        }),
      };
    }


    // ==========================================
    // PHẦN CŨ: CRUD JOURNAL LẤY LẠI ĐỊNH DẠNG RÕ RÀNG
    // ==========================================


    // 4. Route GET /logs
    if (path === "/logs" && method === "GET") {
      const command = new QueryCommand({
        TableName: TABLE_NAME,
        KeyConditionExpression: "userId = :uid",
        ExpressionAttributeValues: { ":uid": userId },
      });
      const response = await docClient.send(command);
     
      const items = await Promise.all(
        (response.Items || []).map(async (item) => {
          const processedImages = await attachSignedUrlsToImages(item.images || []);
          return {
            ...item,
            id: item.entryId,
            images: processedImages,
          };
        })
      );


      return {
        statusCode: 200,
        headers,
        body: JSON.stringify(items),
      };
    }


    // 5. Route POST /logs
    if (path === "/logs" && method === "POST") {
      const body = JSON.parse(event.body || "{}");
      const entryId = body.id || body.entryId || `log_${Date.now()}`;
      const now = new Date().toISOString();


      const item = {
        userId,
        entryId,
        title: body.title || "Chưa có tiêu đề",
        content: body.content || "",
        tags: body.tags || [],
       
        date: body.date || now.split('T')[0],
        startTime: body.startTime || body.start_time || "",
        endTime: body.endTime || body.end_time || "",
        category: body.category || "General",
        commands: body.commands || [],
        errors: body.errors || [],
        solutions: body.solutions || [],
        mood: body.mood !== undefined ? body.mood : "good",
        difficulty: body.difficulty !== undefined ? body.difficulty : "medium",
       
        images: body.images || body.image_keys || [],


        createdAt: body.createdAt || now,
        updatedAt: now,
      };


      const command = new PutCommand({
        TableName: TABLE_NAME,
        Item: item,
      });
      await docClient.send(command);


      const processedImages = await attachSignedUrlsToImages(item.images);


      return {
        statusCode: 201,
        headers,
        body: JSON.stringify({ ...item, id: entryId, images: processedImages }),
      };
    }


    // 6. Route GET /logs/{id}
    if (pathParameters.id && path === `/logs/${pathParameters.id}` && method === "GET") {
      const command = new GetCommand({
        TableName: TABLE_NAME,
        Key: { userId, entryId: pathParameters.id },
      });
      const response = await docClient.send(command);


      if (!response.Item) {
        return { statusCode: 404, headers, body: JSON.stringify({ message: "Không tìm thấy bài viết" }) };
      }


      const item = response.Item;
      const processedImages = await attachSignedUrlsToImages(item.images || []);


      return {
        statusCode: 200,
        headers,
        body: JSON.stringify({
          ...item,
          id: item.entryId,
          images: processedImages
        })
      };
    }


    // 7. Route PUT /logs/{id}
    if (pathParameters.id && path === `/logs/${pathParameters.id}` && method === "PUT") {
      const body = JSON.parse(event.body || "{}");
      const entryId = pathParameters.id;
      const now = new Date().toISOString();


      const item = {
        userId,
        entryId,
        title: body.title,
        content: body.content,
        tags: body.tags || [],
       
        date: body.date,
        startTime: body.startTime || body.start_time,
        endTime: body.endTime || body.end_time,
        category: body.category,
        commands: body.commands || [],
        errors: body.errors || [],
        solutions: body.solutions || [],
        mood: body.mood,
        difficulty: body.difficulty,
       
        images: body.images || body.image_keys || [],


        updatedAt: now,
      };


      const command = new PutCommand({
        TableName: TABLE_NAME,
        Item: item,
      });
      await docClient.send(command);


      const processedImages = await attachSignedUrlsToImages(item.images);


      return { statusCode: 200, headers, body: JSON.stringify({ ...item, id: entryId, images: processedImages }) };
    }


    // 8. Route DELETE /logs/{id}
    if (pathParameters.id && path === `/logs/${pathParameters.id}` && method === "DELETE") {
      const command = new DeleteCommand({
        TableName: TABLE_NAME,
        Key: { userId, entryId: pathParameters.id },
      });
      await docClient.send(command);


      return { statusCode: 200, headers, body: JSON.stringify({ message: "Xóa bài viết thành công" }) };
    }




    // ==========================================
    // PHẦN MỚI: AI REPORTS & QUIZ (SQS ASYNC)
    // ==========================================


    // --- AI REPORTS ROUTES (Giữ nguyên) ---
    if (path === "/ai/reports" && method === "POST") {
      const body = JSON.parse(event.body || "{}");
      const reportId = `rep_${Date.now()}`;
      const now = new Date().toISOString();


      const reportItem = {
        userId,
        reportId,
        type: body.type || "weekly",
        startDate: body.start_date || body.startDate || "",
        endDate: body.end_date || body.endDate || "",
        status: "pending",
        createdAt: now,
      };


      await docClient.send(new PutCommand({ TableName: AI_REPORT_TABLE, Item: reportItem }));


      await sqsClient.send(new SendMessageCommand({
        QueueUrl: AI_REPORT_QUEUE_URL,
        MessageBody: JSON.stringify({ reportId, userId, type: reportItem.type, startDate: reportItem.startDate, endDate: reportItem.endDate }),
      }));


      return {
        statusCode: 202,
        headers,
        body: JSON.stringify({ report_id: reportId, reportId, status: "pending" }),
      };
    }


    if (path === "/ai/reports" && method === "GET") {
      const command = new QueryCommand({
        TableName: AI_REPORT_TABLE,
        KeyConditionExpression: "userId = :uid",
        ExpressionAttributeValues: { ":uid": userId }
      });
      const response = await docClient.send(command);
      return { statusCode: 200, headers, body: JSON.stringify(response.Items || []) };
    }


    if (pathParameters.id && path === `/ai/reports/${pathParameters.id}` && method === "GET") {
      const response = await docClient.send(new GetCommand({
        TableName: AI_REPORT_TABLE,
        Key: { userId, reportId: pathParameters.id }
      }));
      if (!response.Item) return { statusCode: 404, headers, body: JSON.stringify({ message: "Không tìm thấy báo cáo" }) };
      return { statusCode: 200, headers, body: JSON.stringify(response.Item) };
    }


    // --- QUIZ ROUTES (Đã bổ sung dữ liệu cho SQS và API xem lịch sử) ---
   
    // 🌟 ĐÃ BỔ SUNG ROUTE: LẤY DANH SÁCH LỊCH SỬ QUIZ
    if (path === "/quiz" && method === "GET") {
      const command = new QueryCommand({
        TableName: QUIZ_TABLE,
        KeyConditionExpression: "userId = :uid",
        ExpressionAttributeValues: { ":uid": userId }
      });
      const response = await docClient.send(command);
      return { statusCode: 200, headers, body: JSON.stringify(response.Items || []) };
    }


    // --- TẠO QUIZ MỚI ---
    if (path === "/quiz" && method === "POST") {
      const body = JSON.parse(event.body || "{}");
      const quizId = `quiz_${Date.now()}`;
      const now = new Date().toISOString();


      const quizItem = {
        userId,
        quizId,
        sourceType: body.source_type || body.sourceType || "journal",
        topic: body.topic || "General",
        tags: body.tags || [], // Bổ sung tags
        startDate: body.start_date || body.startDate || "", // Bổ sung khoảng thời gian
        endDate: body.end_date || body.endDate || "",
        questionCount: body.question_count || body.questionCount || 5,
        difficulty: body.difficulty || "medium",
        status: "pending",
        createdAt: now,
      };


      await docClient.send(new PutCommand({ TableName: QUIZ_TABLE, Item: quizItem }));


      // Gửi ĐẦY ĐỦ THÔNG TIN cho SQS Worker để nó biết cách lọc nhật ký
      await sqsClient.send(new SendMessageCommand({
        QueueUrl: QUIZ_QUEUE_URL,
        MessageBody: JSON.stringify({
          quizId,
          userId,
          sourceType: quizItem.sourceType,
          topic: quizItem.topic,
          tags: quizItem.tags,
          startDate: quizItem.startDate,
          endDate: quizItem.endDate,
          questionCount: quizItem.questionCount,
          difficulty: quizItem.difficulty
        }),
      }));


      return {
        statusCode: 202,
        headers,
        body: JSON.stringify({ quiz_id: quizId, quizId, status: "pending" }),
      };
    }


    if (pathParameters.id && path === `/quiz/${pathParameters.id}` && method === "GET") {
      const response = await docClient.send(new GetCommand({
        TableName: QUIZ_TABLE,
        Key: { userId, quizId: pathParameters.id }
      }));
      if (!response.Item) return { statusCode: 404, headers, body: JSON.stringify({ message: "Không tìm thấy bài quiz" }) };
      return { statusCode: 200, headers, body: JSON.stringify(response.Item) };
    }


    // API mới: Lấy lịch sử làm bài (Attempts) của một Quiz cụ thể
    if (pathParameters.id && path === `/quiz/${pathParameters.id}/attempts` && method === "GET") {
      const command = new QueryCommand({
        TableName: QUIZ_ATTEMPT_TABLE,
        KeyConditionExpression: "userId = :uid",
        FilterExpression: "quizId = :qid",
        ExpressionAttributeValues: {
          ":uid": userId,
          ":qid": pathParameters.id
        }
      });
      const response = await docClient.send(command);
      return { statusCode: 200, headers, body: JSON.stringify(response.Items || []) };
    }


    if (pathParameters.id && path === `/quiz/${pathParameters.id}/attempts` && method === "POST") {
      const body = JSON.parse(event.body || "{}");
      const quizId = pathParameters.id;
      const attemptId = `att_${Date.now()}`;


      const quizRes = await docClient.send(new GetCommand({
        TableName: QUIZ_TABLE,
        Key: { userId, quizId }
      }));
      if (!quizRes.Item) return { statusCode: 404, headers, body: JSON.stringify({ message: "Không tìm thấy bài quiz" }) };


      const questions = quizRes.Item.questions || [];
      const userAnswers = body.answers || [];
      let score = 0;


      const gradedQuestions = questions.map((q, idx) => {
        // Lấy linh hoạt cả 2 kiểu đặt tên biến
        const correctIdx = q.answer_index ?? q.answerIndex;
        const isCorrect = userAnswers[idx] === correctIdx;
        if (isCorrect) score += 1;
        return { ...q, user_answer: userAnswers[idx], is_correct: isCorrect };
      });


      const attemptItem = {
        userId,
        attemptId,
        quizId,
        score,
        totalQuestions: questions.length,
        answers: gradedQuestions,
        submittedAt: new Date().toISOString(),
      };


      await docClient.send(new PutCommand({ TableName: QUIZ_ATTEMPT_TABLE, Item: attemptItem }));


      return { statusCode: 200, headers, body: JSON.stringify(attemptItem) };
    }


    return {
      statusCode: 404,
      headers,
      body: JSON.stringify({ message: "Route không tồn tại" }),
    };


  } catch (error) {
    console.error("Lỗi Lambda:", error);
    return {
      statusCode: 500,
      headers,
      body: JSON.stringify({ message: "Lỗi xử lý hệ thống", error: error.message }),
    };
  }
};
